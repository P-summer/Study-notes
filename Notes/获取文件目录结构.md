```powershell
# 自定义树形目录函数（可批量排除文件夹）
function Show-Tree {
    param($path='.', $indent=0)
    # ===================== 【可修改区域】在这里添加你要排除的文件夹 =====================
    $exclude = @('node_modules', 'dist', '.git', 'build', 'coverage', 'temp')  # 想用逗号分隔新增
    # ==================================================================================
    
    Get-ChildItem $path | ForEach-Object {
        # 跳过排除的文件夹
        if($_.Name -notin $exclude){
            $line = ("  " * $indent) + "- $($_.Name)"
            Write-Output $line  # 输出到控制台
            if($_.PSIsContainer){ Show-Tree $_.FullName ($indent+1) }
        }
    }
}

# 执行函数 + 导出结果到 项目目录树.txt（自动保存在项目根目录）
Show-Tree | Out-File -FilePath "项目目录树.txt" -Encoding UTF8
```
